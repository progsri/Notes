Debugging

Print all the beans created by Spring Container. 


//Autowire ApplicationContext as dependency as 
@Autowired
ApplicationContext applicationContext;

for(String name : applicationContext.getBeanDefinitionNames() ) {
	try{
		System.out.println(name + " -- " + ApplicationContextProvider.getApplicationContext().getBean(name).toString() ) ;
	}catch(Exception ex){
		System.out.println("ERROR " +  name );
	}
}
