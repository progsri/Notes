Vuex get state


import Vue from "vue";
import Vuex from "vuex";

Vue.use(Vuex); // Should be called before creating the below store object
export const store = new Vuex.Store({
  state: {
    count: 0
  },
  mutations: {
    increment(state) {
      state.count++;
    }
  }
});



<template>
  <div id="app">{{count}}</div>
</template>

<script>
export default {
  data() {
    return {
      count: this.$store.state
    };
  },
  methods: {}
};
</script>
