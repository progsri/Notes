via this.$store.getters 


import Vue from "vue";
import Vuex from "vuex";

Vue.use(Vuex); // Should be called before creating the below store object
export const store = new Vuex.Store({
  state: {
    count: 1
  },
  getters: {
    getTheValue: function(state) {  // getters recieve state as first argument
                                    // array of getters as second argument
      return "The present value is " + state.count;
    }
  }
});



<template>
  <div id="app">{{presentValue()}}</div>
</template>

<script>
export default {
  data() {
    return {
      count: 1
    };
  },
  methods: {
    presentValue() {
      // this.$store.getters gives you access to the getters in the store.
      return this.$store.getters.getTheValue; //do not call like function
    }
  }
};
</script>

</style>
Output 

The present value is 1
