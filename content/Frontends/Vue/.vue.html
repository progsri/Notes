.vue file

props can be access in the child template using {{}} or from the methods as see below. 


import Vue from 'vue'
import App from './App.vue'

new Vue({
  el: '#app',
  render: h => h(App)
})
App.vue 

<template>
  <div>
    {{parent}}
    <child :pass1="parent"></child>
  </div>
</template>

<script>
import Child from "./Child";

export default {
  data() {
    return {
      parent: "This is parent init",
    };
  },
  components: {
    child: Child
  }
};
</script>

<style>
</style>
Child.vue 

<template>
  <div>
    <hr>
    {{child}}
    <br>
    {{pass1}}
    <br>
    <button @click="changeParent">change parent</button>
  </div>
</template>

<script>
export default {
  props: ["pass1"],
  data() {
    return {
      child: " This is child init"
    };
  },
  methods: {
    changeParent: function() {
      this.pass1 = "I can access props"
    }
  }
};
</script>
