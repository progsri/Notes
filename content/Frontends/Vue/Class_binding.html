Class Binding

v-on:class or :class

Binds the css class selector of the element to the data object property of the vue, in this case baseClass

index.html 
    

            .class-gray {
                width: 320px;
                padding: 10px;
                border: 50px solid gray;
                margin: 0;
            }
        
            .class-blue {
                width: 320px;
                padding: 10px;
                border: 50px solid blue;
                margin: 0;
            }

        <div id="app">
        <button v-on:click="changeClass">Change</button>
        <div :class="baseClass">Content</div>
        </div>
main.js 

        var app = new Vue({
            el: '#app',
            data: {
                baseClass: 'class-gray'
            },
            methods: {
                changeClass: function () {
                    if (this.baseClass === 'class-gray') {
                        this.baseClass = 'class-blue'
                    } else {
                        this.baseClass = 'class-gray'
                    }
                }
            }
        }
        )

on Browser

click on the change button changes color
