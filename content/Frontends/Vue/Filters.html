Filters

Lets you apply a function to the value.

Local Filter

This is only visible to that component the filter is defined at 



import Vue from 'vue'
import App from './App.vue'

new Vue({
  el: '#app',
  render: h => h(App)
})
App.vue 

<template>
  <div>
    Value
    <input v-model="value" type="text" />
    <p>After filter :: {{value |  upper}}  </p>
  </div>
</template>

<script>
export default {
  data() {
    return {
      value: ""
    };
  },
  filters:{
    upper: function(entry) {
      return entry.toUpperCase();
    }
  }
};
</script>
